8 JOIN LOCATIONS L ON (L.LOCATION_ID = D.LOCATION_ID)
9 JOIN COUNTRIES C ON (C.COUNTRY_ID = L.COUNTRY_ID)
10 GROUP BY D.DEPARTMENT_NAME, M.LAST_NAME, C.COUNTRY_NAME, M.SALARY*(1+NVL(M.COMMISSION_PCT, 0))
11 HAVING COUNT(E.EMPLOYEE_ID) >= 1;
